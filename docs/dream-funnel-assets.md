// Get variant from URL or assign randomly
function getVariant(testName) {
  var stored = localStorage.getItem('ab_' + testName);
  if(stored) return stored;

  var urlParams = new URLSearchParams(window.location.search);
  var variant = urlParams.get('v') || (Math.random() < 0.5 ? 'a' : 'b');
  localStorage.setItem('ab_' + testName, variant);
  return variant;
}

// Include variant in all events
